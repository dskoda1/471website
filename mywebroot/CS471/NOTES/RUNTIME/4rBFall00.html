The purpose is to illustrate the power of pass by name.  We will use c-like syntax.
 

  • An example of the "power" of pass by name in summing an array is also an example of Jensen's device (a bunch of tricks to operate on arrays)
  • Uses pass by name to apply an operation to an entire array.
int sum (int a, int index,int lower, int upper) {
int temp = 0;
for(index=lower, index<=upper;index ++){
temp = temp+a;
}
return temp;
}
 

 
int n;
int x[ ]=new int[10];
int y[ ]=new int[10];
int i, Ten_ns, xtotal, inner-prod;
Ten_ns = sum(n,i,0,9);
xtotal = sum(x[i],i,0,9);
inner-prod = sum(x[i]*y[i],i,0,9);


sum(n,i,0,9) executes
{   temp=0;
     for(i=0,i<=9,i++){
         temp = temp + n ;
     }
      return temps;
}
  sum(x[i],i,0,9) executes
{   temp=0;
     for(i=0,i<=9,i++){
         temp = temp + x[i] ;
     }
      return temps;
}


sum(x[i]*y[i],i,0,9) executes

{   temp=0;
     for(i=0,i<=9,i++){
         temp = temp + x[i]*y[i] ;
     }
      return temps;
}
